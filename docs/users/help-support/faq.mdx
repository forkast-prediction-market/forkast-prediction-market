import { Accordions, Accordion } from 'fumadocs-ui/components/accordion'

<Accordions type="single" defaultValue="start">
  <Accordion title="How do I start trading?" value="start">
    <p>
      Create an account, deposit USDC, and browse the Markets page on
      {process.env.NEXT_PUBLIC_SITE_NAME}. Prices show the implied probability so
      you can decide which outcome to back.
    </p>
  </Accordion>
  <Accordion title="Where can I review fees before confirming?" value="fees">
    <p>
      The order ticket previews estimated trading fees, expected payout, and gas
      costs prior to signing the transaction in your wallet.
    </p>
  </Accordion>
  <Accordion title="My wallet will not connect" value="wallet">
    <p>
      Make sure your wallet is unlocked, on the Polygon network, and running the
      latest version. Clearing the decentralized app cache usually resolves
      stale sessions.
    </p>
  </Accordion>
  <Accordion title="A transaction is stuck pending" value="pending">
    <p>
      Check the hash on a block explorer. If it remains unmined, speed it up with
      a higher gas fee or cancel and resubmit directly from the wallet controls.
    </p>
  </Accordion>
  <Accordion title="Where do I appeal a resolution?" value="disputes">
    <p>
      Head to the Resolution &amp; Settlement docs for dispute windows and
      evidence requirements. Support can review edge cases when you share the
      market ID and transaction hash.
    </p>
  </Accordion>
  <Accordion title="Need something else?" value="more">
    <p>
      Reach support via live chat, email, or the community channels listed at the
      end of this section. Include relevant transaction details so the team can
      assist quickly.
    </p>
  </Accordion>
</Accordions>
